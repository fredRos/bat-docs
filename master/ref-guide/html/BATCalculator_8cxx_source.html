179 // returns posterior with respect to provided name in the POI ArgSet
180 RooAbsPdf* BATCalculator::GetPosteriorPdf1D(const char* POIname) const
181 {
182  // run some sanity checks
183  if (!fPdf) {
184  std::cerr << "BATCalculator::GetPosteriorPdf - missing pdf model" << std::endl;
185  return 0;
186  }
187 
188  if (!fPrior) {
189  std::cerr << "BATCalculator::GetPosteriorPdf - missing prior pdf" << std::endl;
190  }
191 
192  if (fPOI.getSize() == 0) {
193  std::cerr << "BATCalculator::GetPosteriorPdf - missing parameter of interest" << std::endl;
194  return 0;
195  }
196 
197  if (fPOI.getSize() > 1) {
198  std::cerr << "BATCalculator::GetPosteriorPdf - current implementation works only on 1D intervals" << std::endl;
199  return 0;
200  }
201 
202  //initialize RooInterface object
203  _myRooInterface->Initialize(*fData, *fPdf, *fPrior, fparams, fPOI);
204  _myRooInterface->SetNIterationsRun(_nMCMC);
205  _myRooInterface->MarginalizeAll();
206  _myRooInterface->FindMode();
207  BCH1D myPosterior = _myRooInterface->GetMarginalized(POIname);
208  TH1* posteriorTH1D = myPosterior.GetHistogram();
209  // Is this a memory leak?
210  _posteriorTH1D = static_cast<TH1D*>(BCAux::OwnClone(posteriorTH1D, "_posteriorTH1D"));
211  {
212  BCAux::RootSideEffectGuard g;
213  RooDataHist* posteriorRooDataHist = new RooDataHist("posteriorhist", "", fPOI, posteriorTH1D);
214  fPosteriorPdf = new RooHistPdf("posteriorPdf", "", fPOI, *posteriorRooDataHist);
215  }
216 
217  return fPosteriorPdf; // is of type RooAbsPdf*
218 }

258 // returns central interval for the defined POI in the POI ArgSet->test code because orginal version is not working anymore
259 SimpleInterval* BATCalculator::GetInterval1D(const char* POIname) const
260 {
261  //const char * POIname = fPOI.first()->GetName();
262 
263  if (fValidInterval)
264  std::cout << "BATCalculator::GetShortestInterval1D:"
265  << "Warning : recomputing interval for the same CL and same model" << std::endl;
266 
267  // get pointer to selected parameter of interest
268  RooRealVar* poi = dynamic_cast<RooRealVar*>( fPOI.find(POIname) );
269  assert(poi);
270 
271  // get pointer to poserior pdf
272  if (!fPosteriorPdf)
273  fPosteriorPdf = (RooAbsPdf*) GetPosteriorPdf1D();
274  //RooAbsReal * cdf = fPosteriorPdf->createCdf(fPOI,RooFit::ScanParameters(100,2));
275 
276  // range of poi
277  Double_t minpoi = poi->getMin();
278  Double_t maxpoi = poi->getMax();
279 
280  // bin number of histogram for posterior
281  Int_t stepnumber = _posteriorTH1D->GetNbinsX();
282 
283  // width of one bin in histogram of posterior
284  Double_t stepsize = (maxpoi - minpoi) / stepnumber;
285 
286  // pair: first entry: bin number , second entry: value of posterior
287  std::vector< std::pair< Int_t, Double_t > > posteriorVector;
288 
289  // for normalization
290  Double_t histoIntegral = 0;
291  // give posteriorVector the right length
292  posteriorVector.resize(stepnumber);
293 
294  // initialize elements of posteriorVector
295  int i = 0;
296  std::vector< std::pair< Int_t, Double_t > >::iterator vecit = posteriorVector.begin();
297  std::vector< std::pair< Int_t, Double_t > >::iterator vecit_end = posteriorVector.end();
298  for ( ; vecit != vecit_end ; ++vecit) {
299  poi->setVal(poi->getMin() + i * stepsize);
300  posteriorVector[i] = std::make_pair(i, _posteriorTH1D->GetBinContent(i + 1) ); // hope this is working, +1 necessary, because GetBinContent(0) returns the underflow bin
301  histoIntegral += _posteriorTH1D->GetBinContent(i); // better to get this directly from the histogram ?!
302  ++i;
303  }
304 
305  double lowerProbLim = (1. - ConfidenceLevel()) * fLeftSideFraction;
306  double upperProbLim = 1. - ( (1. - ConfidenceLevel()) * (1. - fLeftSideFraction) );
307 
308  fLower = -1.;
309  fUpper = -1.;
310  bool lowerlimset = false;
311  bool upperlimset = false;
312 
313 
314  if (fLeftSideFraction == 0) {
315  fLower = minpoi;
316  lowerlimset = true;
317  }
318 
319  if (fLeftSideFraction == 1) {
320  fUpper = maxpoi;
321  upperlimset = true;
322  }
323 
324  // keep track of integrated posterior in the interval
325  Double_t integratedposterior = 0.;
326 
327  i = 0;
328  vecit = posteriorVector.begin();
329  for ( ; vecit != vecit_end ; ++vecit) {
330  integratedposterior += posteriorVector[i].second;
331  if ( (lowerlimset != true) && ((integratedposterior / histoIntegral) >= lowerProbLim) ) {
332  fLower = poi->getMin() + i * stepsize;
333  lowerlimset = true;
334  }
335  if ( (upperlimset != true) && ((integratedposterior / histoIntegral) >= upperProbLim) ) {
336  fUpper = poi->getMin() + i * stepsize;
337  upperlimset = true;
338  break;
339  }
340  i++;
341  }
342 
343  TString interval_name = TString("CentralBayesianInterval_a") + TString(this->GetName());
344  SimpleInterval* interval = new SimpleInterval(interval_name, *poi, fLower, fUpper, ConfidenceLevel());
345  interval->SetTitle("SimpleInterval from BATCalculator");
346 
347  return interval;
348

359 // ---------------------------------------------------------
360 // returns a SimpleInterval with lower and upper bounds on the
361 // parameter of interest. Applies the shortest interval rule to
362 // compute the credibility interval. The resulting interval is not necessarily connected.
363 // Methods for constructing the shortest region with respect given CL are now available in
364 // different places (here; MCMCCalculator, BAT)-> this might require cleaning at some point.
365 // Result is approximate as CL is not reached exactly (due to finite bin number/bin size in posterior histogram)-> make CL/interval a bit smaller or bigger ?
366 SimpleInterval* BATCalculator::GetShortestInterval1D(const char* POIname, bool& checkConnected) const
367 {
368 
369  if (fValidInterval)
370  std::cout << "BATCalculator::GetShortestInterval1D:"
371  << "Warning : recomputing interval for the same CL and same model" << std::endl;
372 
373  // get pointer to selected parameter of interest
374  RooRealVar* poi = dynamic_cast<RooRealVar*>( fPOI.find(POIname) );
375  assert(poi);
376 
377  // get pointer to poserior pdf
378  if (!fPosteriorPdf)
379  fPosteriorPdf = (RooAbsPdf*) GetPosteriorPdf1D();
380 
381  // range of poi
382  Double_t minpoi = poi->getMin();
383  Double_t maxpoi = poi->getMax();
384 
385  // bin number of histogram for posterior
386  Int_t stepnumber = _posteriorTH1D->GetNbinsX();
387 
388  // width of one bin in histogram of posterior
389  Double_t stepsize = (maxpoi - minpoi) / stepnumber;
390 
391  // pair: first entry: bin number , second entry: value of posterior
392  std::vector< std::pair< Int_t, Double_t > > posteriorVector;
393 
394  // for normalization
395  Double_t histoIntegral = 0;
396  // give posteriorVector the right length
397  posteriorVector.resize(stepnumber);
398 
399  // see in BayesianCalculator for details about this "feature"
400  Double_t tmpVal = poi->getVal();
401 
402  // initialize elements of posteriorVector
403  int i = 0;
404  std::vector< std::pair< Int_t, Double_t > >::iterator vecit = posteriorVector.begin();
405  std::vector< std::pair< Int_t, Double_t > >::iterator vecit_end = posteriorVector.end();
406  for ( ; vecit != vecit_end ; ++vecit) {
407  poi->setVal(poi->getMin() + i * stepsize);
408  posteriorVector[i] = std::make_pair(i, _posteriorTH1D->GetBinContent(i + 1) ); // hope this is working, +1 necessary, because GetBinContent(0) returns the underflow bin
409  histoIntegral += _posteriorTH1D->GetBinContent(i); // better to get this directly from the histogram ?!
410  i++;
411  }
412 
413  // sort posteriorVector with respect to posterior pdf
414  std::sort(posteriorVector.begin(), posteriorVector.end(), ::sortbyposterior);
415 
416  // keep track of integrated posterior in the interval
417  Double_t integratedposterior = 0.;
418 
419  // keep track of lowerLim and upperLim
420  Double_t lowerLim = posteriorVector.size();
421  Double_t upperLim = 0;
422 
423  // store the bins in the intervall
424  std::vector<bool> inInterval;
425  inInterval.resize(posteriorVector.size());
426 
427  // set all values in inInterval to false
428  for (unsigned int k = 0; k < inInterval.size(); k++)
429  inInterval[k] = false;
430 
431  unsigned int j = 0;
432  // add bins to interval while CL not reached
433  while (((integratedposterior / histoIntegral) < (1 - fSize)) && (j < posteriorVector.size())) {
434  integratedposterior += posteriorVector[j].second;
435 
436  // update vector with bins included in the interval
437  inInterval[posteriorVector[j].first] = true;
438 
439  if (posteriorVector[j].first < lowerLim) {
440  lowerLim = posteriorVector[j].first; // update lowerLim
441  }
442  if (posteriorVector[j].first > upperLim) {
443  upperLim = posteriorVector[j].first; // update upperLim
444  }
445 
446  fLower = lowerLim * stepsize; //
447  fUpper = upperLim * stepsize;
448  j++;
449  }
450 
451  // initialize vector with interval borders
452 
453  bool runInside = false;
454  for (unsigned int l = 0; l < inInterval.size(); l++) {
455  if ( (runInside == false) && (inInterval[l] == true) ) {
456  _intervalBorders1D.push_back(static_cast<double>(l)* stepsize);
457  runInside = true;
458  }
459  if ( ( runInside == true) && (l < (inInterval.size() - 1) ) && (inInterval[l + 1] == false) ) {
460  _intervalBorders1D.push_back(static_cast<double>(l)* stepsize);
461  runInside = false;
462  }
463  if ( ( runInside == true) && (l == (inInterval.size() - 1)) ) {
464  _intervalBorders1D.push_back(static_cast<double>(l)* stepsize);
465  }
466  }
467 
468  // check if the intervall is connected
469  if (checkConnected) {
470  if (_intervalBorders1D.size() > 2) {
471  fConnectedInterval = false;
472  } else {
473  fConnectedInterval = true;
474  }
475  }
476 
477  poi->setVal(tmpVal); // patch: restore the original value of poi
478 
479  fValidInterval = true;
480 
481  TString interval_name = TString("ShortestBayesianInterval_a") + TString(this->GetName());
482  SimpleInterval* interval = new SimpleInterval(interval_name, *poi, fLower, fUpper, ConfidenceLevel());
483  interval->SetTitle("Shortest SimpleInterval from BATCalculator");
484 
485  //if(assert)
486  return interval;
487 }
