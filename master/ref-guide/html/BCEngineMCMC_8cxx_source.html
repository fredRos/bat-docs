269 void BCEngineMCMC::CloneMarginals(const BCEngineMCMC& other)
270 {
271  fH1Marginalized = std::vector<TH1*>(other.fH1Marginalized.size(), NULL);
272  for (unsigned i = 0; i < other.fH1Marginalized.size(); ++i)
273  if (other.fH1Marginalized[i])
274  fH1Marginalized[i] = BCAux::OwnClone(other.fH1Marginalized[i]);
275 
276  if (!other.fH2Marginalized.empty() && !other.fH2Marginalized.front().empty()) {
277  fH2Marginalized = std::vector<std::vector<TH2*> > (other.fH2Marginalized.size(), std::vector<TH2*>(other.fH2Marginalized.front().size(), NULL));
278  for (unsigned i = 0; i < other.fH2Marginalized.size(); ++i) {
279  fH2Marginalized[i].assign(other.fH2Marginalized[i].size(), NULL);
280  for (unsigned j = 0; j < other.fH2Marginalized[i].size(); ++j)
281  if (other.fH2Marginalized[i][j])
282  fH2Marginalized[i][j] = BCAux::OwnClone(other.fH2Marginalized[i][j]);
283  }
284  }
285 }

384  fMCMCNIterationsPreRunMax = 100000;

464 void BCEngineMCMC::WriteMarginalizedDistributions(const std::string& filename, const std::string& option, bool closeExistingFile)
465 {
466  // remember current directory
467  TDirectory* dir = gDirectory;
468 
469  // look to see if file is already open
470  TSeqCollection* listOfFiles = gROOT->GetListOfFiles();
471  TFile* fOut = NULL;
472  for (int i = 0; i < listOfFiles->GetEntries(); ++i)
473  if (listOfFiles->At(i) && filename.compare(listOfFiles->At(i)->GetName()) == 0) {
474  fOut = dynamic_cast<TFile*>(listOfFiles->At(i));
475  break;
476  }
477  if (fOut) {
478  if (option.compare("RECREATE") == 0) {
479  // if recreating, close current file, which will be overwritten
480  BCLog::OutError("BCEngineMCMC::WriteMarginalizedDistributions: File already open. option \"RECREATE\" will now overwrite it!");
481  if (fOut->IsWritable())
482  fOut->Write(0, TObject::kWriteDelete);
483  fOut->Close();
484  fOut = NULL;
485  closeExistingFile = true; // existing file closed; close newly opened file
486  } else if (option.compare("UPDATE") == 0 && !fOut->IsWritable()) {
487  BCLog::OutError("BCEngineMCMC::WriteMarginalizedDistributions: File already open but not in readable mode.");
488  return;
489  }
490  } else {
491  closeExistingFile = true; // no pre-open file found; close newly opened file
492  }
493 
494  // else open file
495  if (!fOut)
496  fOut = TFile::Open(filename.c_str(), option.c_str());
497 
498  if (!fOut) {
499  BCLog::OutError("BCEngineMCMC::WriteMarginalizedDistributions: Could not open output file.");
500  return;
501  }
502 
503  if (!fOut->IsWritable()) {
504  BCLog::OutError("BCEngineMCMC::WriteMarginalizedDistributions: File must be opened in writeable mode.");
505  return;
506  }
507 
508  // write histograms
509  for (unsigned i = 0; i < GetNVariables(); ++i) {
510  if (MarginalizedHistogramExists(i))
511  fOut->WriteTObject(GetMarginalizedHistogram(i));
512  for (unsigned j = 0; j < GetNVariables(); ++j)
513  if (MarginalizedHistogramExists(i, j))
514  fOut->WriteTObject(GetMarginalizedHistogram(i, j));
515  }
516 
517  // Causes multiple instances of any tree that already exists in the root file.
518  // fOut->Write();
519 
520  if (closeExistingFile)
521  fOut->Close();
522 
523  // restore directory
524  gDirectory = dir;
525 }

545 // --------------------------------------------------------
546 TH2* BCEngineMCMC::GetMarginalizedHistogram(unsigned i, unsigned j) const
547 {
548  if (i == j) {
549  BCLog::OutError(Form("BCEngineMCMC::GetMarginalizedHistogram. Called with identical indices %u.", i));
550  return NULL;
551  }
552 
553  if (i >= fH2Marginalized.size()) {
554  BCLog::OutError(Form("BCEngineMCMC::GetMarginalizedHistogram. Index %u out of range.", i));
555  return NULL;
556  }
557  if (j >= fH2Marginalized[i].size()) {
558  BCLog::OutError(Form("BCEngineMCMC::GetMarginalizedHistogram. Index %u out of range.", j));
559  return NULL;
560  }
561 
562  if (fH2Marginalized[i][j])
563  return fH2Marginalized[i][j];
564

1036  partree->SetBranchAddress("fixed_value", &p_fixedvalue);
1037  else
1038  has_fixed_value = false;

1041  for (unsigned i = 0; i < GetNParameters(); ++i) {
1042  if (partree->GetEntryNumberWithIndex(1, i) < 0) {
1043  BCLog::OutError("BCEngineMCMC::ParameterTreeMatchesModel : Parameter tree contains too few entries.");
1044  return false;
1045  }
1046  partree->GetEntryWithIndex(1, i);
1047  if (!GetParameter(i).IsNamed(p_name)) {
1048  BCLog::OutError(Form("BCEngineMCMC::ParameterTreeMatchesModel : Parameter[%d]'s names do not match.", i));

1054  BCLog::OutWarning(Form("BCEngineMCMC::ParameterTreeMatchesModel : Upper limit of parameter \"%s\" does not match.", p_name));
1055  if (has_fixed && GetParameter(i).Fixed() != p_fixed) {
1056  BCLog::OutWarning(Form("BCEngineMCMC::ParameterTreeMatchesModel : Fixed status of parameter \"%s\" does not match. Fixing it.", p_name));
1057  GetParameter(i).Fix(p_fixedvalue);
1058  }
1059  if (has_fixed && GetParameter(i).Fixed() && has_fixed_value && GetParameter(i).GetFixedValue() != p_fixedvalue) {
1060  BCLog::OutWarning(Form("BCEngineMCMC::ParameterTreeMatchesModel : Fixed value of parameter \"%s\" does not match. Updating it.", p_name));
1061  GetParameter(i).Fix(p_fixedvalue);
1062  }
1063  }
1064  if (!checkObservables)
1065  return true;
1066  // check observables
1067  for (unsigned i = 0; i < GetNObservables(); ++i) {
1068  if (partree->GetEntryNumberWithIndex(0, i) < 0) {
1069  BCLog::OutError("BCEngineMCMC::ParameterTreeMatchesModel : Parameters tree contains too few entries.");
1070  return false;
1071  }
1072  partree->GetEntryWithIndex(0, i);
1073  if (!GetObservable(i).IsNamed(p_name)) {
1074  BCLog::OutError(Form("BCEngineMCMC::ParameterTreeMatchesModel : Observable[%d]'s names do not match.", i));
1075  return false;
1076  }
1077  if (GetObservable(i).GetLowerLimit() != p_lowerlimit)
1078  BCLog::OutWarning(Form("BCEngineMCMC::ParameterTreeMatchesModel : Lower limit of observable \"%s\" does not match.", p_name));
1079  if (GetObservable(i).GetUpperLimit() != p_upperlimit)
1080  BCLog::OutWarning(Form("BCEngineMCMC::ParameterTreeMatchesModel : Upper limit of observable \"%s\" does not match.", p_name));
1081  }
1082  return true;
1083 }

1086 bool BCEngineMCMC::LoadMCMC(const std::string& filename, const std::string& mcmcTreeName, const std::string& parameterTreeName, bool loadObservables)
1087 {

1095  }
1096 
1097  // set model name if empty
1098  if (fName.empty()) {
1099  // check mcmcTreeName and parameterTreeName for default BAT name scheme [modelname]_mcmc/parameters:
1100  if (mcmcTreeName.find_last_of("_") != std::string::npos && mcmcTreeName.substr(mcmcTreeName.find_last_of("_")) == "_mcmc"
1101  && parameterTreeName.find_last_of("_") != std::string::npos && parameterTreeName.substr(parameterTreeName.find_last_of("_")) == "_parameters") {
1102  fName = mcmcTreeName.substr(0, mcmcTreeName.find_last_of("_"));
1103  }
1104  // else look through file for trees named according to BAT scheme
1105  else {
1106  TList* LoK = inputfile->GetListOfKeys();
1107  std::vector<std::string> mcmc_names;
1108  std::vector<std::string> parameter_names;
1109  for (int i = 0; i < LoK->GetEntries(); ++i) {
1110  TKey* k = (TKey*)(LoK->At(i));
1111  if (strcmp(k->GetClassName(), "TTree") != 0)
1112  continue;
1113  std::string treeName(k->GetName());
1114  if (treeName.find_last_of("_") == std::string::npos)
1115  continue;
1116  if (treeName.substr(treeName.find_last_of("_")) == "_mcmc")

1370 
1371  // a = (1+t)^(-lambda)

1385  fMultivariateProposalFunctionCovariance[c][I][J] += a * fMCMCStatistics[c].covariance[i][j];
1386  fMultivariateProposalFunctionCovariance[c][J][I] = fMultivariateProposalFunctionCovariance[c][I][J];

2019 double BCEngineMCMC::RValue(const std::vector<double>& means, const std::vector<double>& variances, unsigned n, bool correctForSamplingVariability)
2020 {
2021  // calculate R values according to Brooks & Gelman,
2022  // "General Methods for Monitoring Convergence of Iterative Simulations, 1998
2023 
2024  if (means.empty() || variances.empty() || means.size() != variances.size() || n == 0)
2025  return std::numeric_limits<double>::quiet_NaN();
2026 
2027  unsigned m = means.size();
2028 
2029  // means of values
2030  double mean_of_means = 0;
2031  double mean_of_variances = 0;
2032  for (unsigned c = 0; c < m; ++c) {
2033  mean_of_means += means[c];
2034  mean_of_variances += variances[c];
2035  }
2036  mean_of_means *= 1. / m;
2037  mean_of_variances *= 1. / m;
2038 
2039  // variances of values
2040  double variance_of_means = 0;
2041  double variance_of_variances = 0;
2042  for (unsigned c = 0; c < m; ++c) {
2043  variance_of_means += (means[c] - mean_of_means) * (means[c] - mean_of_means);
2044  variance_of_variances += (variances[c] - mean_of_variances) * (variances[c] - mean_of_variances);
2045  }
2046  variance_of_means /= m - 1.;
2047  variance_of_variances /= m - 1.;
2048 
2049  // variance of all samples:
2050  double full_variance = m * (n - 1.) / (m * n - 1) * mean_of_variances + n * (m - 1.) / (m * n - 1) * variance_of_means;
2051 
2052  if (mean_of_variances == 0) {
2053  BCLOG_DEBUG("mean of variances is zero!");
2054  if (full_variance == 0) {
2055  BCLOG_DEBUG("variance of all samples is also zero!");
2056  return 1;
2057  } else
2058  return std::numeric_limits<double>::infinity();
2059  }
2060 
2061  double rvalue = sqrt(full_variance / mean_of_variances); // variance(all samples) / mean(chain variances)
2062 
2063  if (!correctForSamplingVariability)
2064  return rvalue;
2065 
2066  // else correct for initial sampling variability
2067 
2068  double meansquare_of_means = 0;
2069  for (unsigned c = 0; c < m; ++c)
2070  meansquare_of_means += means[c] * means[c];
2071  meansquare_of_means *= 1. / m;
2072 
2073  double covariance_of_variance_with_mean = 0;
2074  double covariance_of_variance_with_meansquare = 0;
2075  for (unsigned c = 0; c < m; ++c) {
2076  covariance_of_variance_with_mean += (variances[c] - mean_of_variances) * (means[c] - mean_of_means);
2077  covariance_of_variance_with_meansquare += (variances[c] - mean_of_variances) * (means[c] * means[c] - meansquare_of_means);
2078  }
2079  covariance_of_variance_with_mean /= m - 1.;
2080  covariance_of_variance_with_meansquare /= m - 1.;
2081 
2082  double N = (n - 1.) / n;
2083  double M = (m + 1.) / m;
2084 
2085  double V = N * full_variance + M * variance_of_means;
2086 
2087  double varV = N * N / m * variance_of_variances + 2 * M / (m - 1) * variance_of_means * variance_of_means
2088  + 2 * M * N / m * (covariance_of_variance_with_meansquare - 2 * mean_of_means * covariance_of_variance_with_mean);
2089 
2090  double df = 2 * V * V / varV;
2091 
2092  return rvalue * sqrt((df + 3) / (df + 1));
2093 }

3755 void BCEngineMCMC::Statistics::Update(const ChainState& cs)
3756 {
3757  if (mean.size() != cs.parameters.size() + cs.observables.size())
3758  return;
3759 
3760  // increment number of samples
3761  ++n_samples;
3762 
3763  // check mode
3764  if (cs.log_probability > probability_at_mode) {
3765  modepar = cs.parameters;
3766  modeobs = cs.observables;
3767  probability_at_mode = cs.log_probability;
3768  }
3769 
3770  // update probability mean and variance
3771  double prob_delta = cs.log_probability - probability_mean;
3772  probability_mean += prob_delta / n_samples;
3773  probability_variance += (n_samples > 1) ? prob_delta * prob_delta / n_samples - probability_variance / (n_samples - 1) : 0;
3774 
3775  // update parameter means and (co)variances, and maxima and minima:
3776 
3777  // vector to store difference from current mean
3778  std::vector<double> delta(mean.size(), 0);
3779 
3780  // loop over values
3781  for (unsigned i = 0; i < mean.size(); ++i) {
3782  // get value
3783  double x = (i < cs.parameters.size()) ? cs.parameters[i] : cs.observables[i - cs.parameters.size()];
3784  // store difference to current mean
3785  delta[i] = x - mean[i];
3786  // update mean
3787  mean[i] += delta[i] / n_samples;
3788  // update variance
3789  variance[i] += (n_samples > 1) ? delta[i] * delta[i] / n_samples - variance[i] / (n_samples - 1.) : 0;
3790  // update minimum & maximum
3791  minimum[i] = std::min(minimum[i], x);
3792  maximum[i] = std::max(maximum[i], x);
3793  }
3794  // update covariances
3795  if (n_samples > 1) {
3796  for (unsigned i = 0; i < mean.size(); ++i)
3797  for (unsigned j = i; j < mean.size(); ++j)
3798  covariance[i][j] += delta[i] * delta[j] / n_samples - covariance[i][j] / (n_samples - 1);
3799  }
3800  // update stderrors
3801  for (unsigned i = 0; i < modepar.size(); ++i)
3802  stderrpar[i] = std::sqrt(variance[i]);
3803  for (unsigned i = 0; i < modeobs.size(); ++i)
3804  stderrobs[i] = std::sqrt(variance[stderrpar.size() + i]);
3805 }

3813

3849  }

3852  stderrpar[i] = std::sqrt(variance[i]);
3853  for (unsigned i = 0; i < stderrobs.size(); ++i)
