81 void BCParameter::SetLimits(double lowerlimit, double upperlimit)
82 {
83  // fixed automatically due to identical limits before?
84  if (GetRangeWidth() == 0.0 && Fixed())
85  Unfix();
86 
87  BCVariable::SetLimits(lowerlimit, upperlimit);
88  if (BCAux::RangeType(fLowerLimit, fUpperLimit) == BCAux::kFiniteRange and fPrior)
89  fPrior->GetFunction().SetRange(fLowerLimit, fUpperLimit);
90  if (lowerlimit == upperlimit)
91  Fix(lowerlimit);
92 }
